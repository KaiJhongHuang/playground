☁️ Google Sheets 雲端同步

讓資料在不同裝置間同步。使用者自己部署 Google Apps Script，網址只存在此裝置的瀏覽器，不會上傳到網路。

📖 設定步驟
  1. 建立一份新的 Google 試算表
  2. 選單 → 擴充功能 → Apps Script
  3. 貼上下方程式碼（取代原有全部內容）並儲存
  4. 點右上角「部署」→「新增部署作業」
  5. 類型：網頁應用程式，執行身分：我，存取權：所有人
  6. 點「部署」後複製產生的網址
  7. 貼到下方的「同步網址」欄位
function doGet(e) {
  var sheet = _sheet();
  var data = sheet.getRange('A1').getValue();
  return ContentService.createTextOutput(data || '{}')
    .setMimeType(ContentService.MimeType.JSON);
}
function doPost(e) {
  var body = JSON.parse(e.postData.contents);
  var sheet = _sheet();
  sheet.getRange('A1').setValue(JSON.stringify(body.data));
  sheet.getRange('A2').setValue(new Date().toISOString());
  return ContentService.createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
function _sheet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var s = ss.getSheetByName('HeroGrowth');
  if (!s) s = ss.insertSheet('HeroGrowth');
  return s;
}
